  4 /**
  5  * classe di base delle procedure di controllo di una applicazione waLibs.
  6  * 
  7  * L'XSLT di default, una volta caricato questo file,  cercherà di caricare un 
  8  * modulo (file) javascript con nome [nome_pagina].js. Dopodichè verificherà se
  9  * esiste già un oggetto in document.wapagina . Se non esiste, creerà quell'oggetto
 10  * instanziando la presente classe.
 11  * 
 12  * E' così possibile per il programmatore estendere la presente classe nel file 
 13  * [nome_pagina].js e sempre all'interno del file istanziarne l'estensione in 
 14  * document.wapagina.
 15  * 
 16  * I compiti principali della classe sono:
 17  * 
 18  * <ul>
 19  * <li>gestire la navigazione tra le pagine (apertura/chiusura/aggiornamento)
 20  * <li>collegare le azioni di eventuali oggetti waTabella presenti nella pagina
 21  *		con metodi implementati  all'interno dell'estensione della classe
 22  *		(ossia in [nome_pagina].js). Si veda al proposito la nota in 
 23  *		{@link collegaAzioniTabella}
 24  * <li>collegare i link di eventuali oggetti waTabella presenti nella pagina
 25  *		con metodi implementati  all'interno dell'estensione della classe
 26  *		(ossia in [nome_pagina].js). Si veda al proposito la nota in 
 27  *		{@link collegaLinkTabella}
 28  * <li>collegare gli eventi di eventuali oggetti contenuti nei waModuli presenti 
 29  *		nella pagina
 30  *		con metodi implementati  all'interno dell'estensione della classe
 31  *		(ossia in [nome_pagina].js). Si veda al proposito la nota in 
 32  *		{@link collegaEventiModulo}.
 33  *	</ul>
 34  *	
 35  *	Naturalmente è poi possibile che [nome_pagina].js dichiari una classe che
 36  *	non deriva direttamente da waapplicazione, ma da una classe intermedia che
 37  *	derivi da waapplicazione, e che implementi metodi personalizzati per il 
 38  *	funzionamento della propria applicazione (calcolo codice fiscale, gestione
 39  *	help online, ecc.). Difatti l'XSLT di default cercherà anche di caricare un
 40  *	modulo (file) javascript con nome [nome_applicazione].js
 41  *	
 42  *	@class waapplicazione

229 	/**
230 	 * metodo documentato a soli fini illustrativi, ma che non ha alcuna 
231 	 * necessità di essere invocato dal programmatore, in quanto richiamato
232 	 * dalla classe in fase di inizializzazione.
233 	 * 
234 	 * Questo metodo cerca eventuali metodi definiti all'interno della classe
235 	 * (e naturalmente sue derivate) che abbiano una corrispondenza con una
236 	 * delle azioni definite all'interno di un oggetto watabella. Se trova la
237 	 * corrispondenza, associa all'azione della tabella il metodo della classe,
238 	 * tale per cui quando l'utente richiede l'azione viene invocato il metodo
239 	 * della classe. Inoltre pone lo scope del metodo all'interno della classe
240 	 * derivata da waapplicazione, anzichè di watabella.
241 	 * <p>
242 	 * Ad esempio: supponiamo di avere nella pagina un oggetto watabella il cui
243 	 * nome è <b>tabella_ordini</b>, e che per quella tabella sia stato richiesto di
244 	 * implementare l'azione su riga <b>archivia</b>. 
245 	 * </p>
246 	 * Supponiamo ancora che abbiate istanziato in document.wapagina un oggetto
247 	 * di classe derivata da waapplicazione e che implementi il metodo
248 	 * </p>
249 	 * <pre>
250 	 * azione_tabella_ordini_archivia : function(idRiga)
251 	 * </pre>
252 	 * <p>
253 	 * Ebbene, ogni volta che l'utente premerà il bottone (o qualsiasi altro
254 	 * meccanismo) "archivia" verrà invocato il metodo da voi definito in 
255 	 * document.wapagina, e lo scope del metodo sarà quello di document.wapagina,
256 	 * ossia <b>this</b> corrisponderà a document.wapagina e non a
257 	 * document.tabella_ordini.
258 	 * </p>
259 	 * <p>
260 	 * A questo punto voi potrete chiamare il vostro server avvertendolo che
261 	 * l'utente ha richiesto l'archiviazione della riga; ad esempio (supponendo
262 	 * che la riga ordine abbia anche una colonna con nome <b>titolo</b>):
263 	 * </p>
264 	 * <pre>
265 	 * {
266 	 * if (confirm("Confermi archiviazione ordine " + this.tabelle["tabella_ordini"].righe[idRiga].campi.titolo + "?"))
267 	 *	location.href = "?archivia=1&idRiga=" + idRiga;
268 	 * }
269 	 * </pre>
270 	 */

272 		{
273 		var nomeMetodo;
274 		var nomeTabella;
275 		
276 		for (nomeMetodo in this)
277 			{
278 			if (typeof(this[nomeMetodo]) == 'function' && nomeMetodo.substr(0, ("azione_").length) == "azione_")
279 				{
280 				for (nomeTabella in this.tabelle)
281 					{
282 					if (nomeMetodo.substr(0, ("azione_" + nomeTabella + "_").length) == "azione_" + nomeTabella + "_")
283 						this.tabelle[nomeTabella][nomeMetodo] = new Function ("id", "return this.applicazione." + nomeMetodo + "(id)");
284 					}
285 				}
286 			}

290 	/**
291 	 * metodo documentato a soli fini illustrativi, ma che non ha alcuna 
292 	 * necessità di essere invocato dal programmatore, in quanto richiamato
293 	 * dalla classe in fase di inizializzazione.
294 	 * 
295 	 * Questo metodo cerca eventuali metodi definiti all'interno della classe
296 	 * (e naturalmente sue derivate) che abbiano una corrispondenza con uno
297 	 * dei link definiti all'interno di un oggetto watabella. Se trova la
298 	 * corrispondenza, associa al link della tabella il metodo della classe,
299 	 * tale per cui quando l'utente segue il link viene invocato il metodo
300 	 * della classe. Inoltre pone lo scope del metodo all'interno della classe
301 	 * derivata da waapplicazione, anzichè di watabella.
302 	 * <p>
303 	 * Ad esempio: supponiamo di avere nella pagina un oggetto watabella il cui
304 	 * nome è <b>tabella_ordini</b>, e che per quella tabella sia stato richiesto di
305 	 * implementare un link sulla colonna <b>referente</b>. 
306 	 * </p>
307 	 * Supponiamo ancora che abbiate istanziato in document.wapagina un oggetto
308 	 * di classe derivata da waapplicazione e che implementi il metodo
309 	 * </p>
310 	 * <pre>
311 	 * link_tabella_ordini_referente : function(idRiga)
312 	 * </pre>
313 	 * <p>
314 	 * Ebbene, ogni volta che l'utente seguirà il link nella cella della colonna
315 	 * <b>referente</b> verrà invocato il metodo da voi definito in 
316 	 * document.wapagina, e lo scope del metodo sarà quello di document.wapagina,
317 	 * ossia <b>this</b> corrisponderà a document.wapagina e non a
318 	 * document.tabella_ordini.
319 	 * </p>
320 	 * <p>
321 	 * A questo punto voi potrete, ad esempio, invocare il client di posta 
322 	 * elettronica passandogli l'indirizzo email del referente (che avrete 
323 	 * ovviamente inserito in una delle colonne della tabella, magari non visibile):
324 	 * </p>
325 	 * <pre>
326 	 * {
327 	 * location.href = "mailto:" + this.tabelle["tabella_ordini"].righe[id].campi.EmailReferente;
328 	 * }
329 	 * </pre>
330 	 */

332 		{
333 		var nomeMetodo;
334 		var nomeTabella;
335 		
336 		for (nomeMetodo in this)
337 			{
338 			if (typeof(this[nomeMetodo]) == 'function' && nomeMetodo.substr(0, ("link_").length) == "link_")
339 				{
340 				for (nomeTabella in this.tabelle)
341 					{
342 					if (nomeMetodo.substr(0, ("link_" + nomeTabella + "_").length) == "link_" + nomeTabella + "_")
343 						this.tabelle[nomeTabella][nomeMetodo] = new Function ("id", "return this.applicazione." + nomeMetodo + "(id)");
344 					}
345 				}

350 	// funzione richiamata automaticamente dalla collegaEventiModulo
351 	eventoApplicazione: function (event)
352 		{
353 		// siamo nello scope dell'oggetto fisico che ha innescato l'evento;
354 		// vogliamo andare nello scope di questo (this) oggetto;
355 		// cerchiamo qual'e' il controllo a cui appartiene l'oggetto che ha
356 		// scatenato l'evento, in modo da ricostruire il nome del metodo da
357 		// richiamare
358 		
359 		var nomeMetodo = "evento_on" + (window.event || event).type + "_";
360 		
361 		// verifichiamo se non sia un evento del modulo
362 		if (this.elements)
363 			return this.wamodulo.applicazione[nomeMetodo + this.id + "_" + this.id](event);
364 
365 		var nomeControllo = '';
366 		for (nomeControllo in this.form.wamodulo.controlli)
367 			{
368 			if (this.form.wamodulo.controlli[nomeControllo].miAppartiene(this))
369 				break;
370 			}
371 			
372 		nomeMetodo += this.form.id + "_" + this.form.wamodulo.controlli[nomeControllo].nome;
373 		return this.form.wamodulo.applicazione[nomeMetodo](event);
374 		},

377 	/**
378 	 * metodo documentato a soli fini illustrativi, ma che non ha alcuna 
379 	 * necessità di essere invocato dal programmatore, in quanto richiamato
380 	 * dalla classe in fase di inizializzazione.
381 	 * 
382 	 * Questo metodo cerca eventuali metodi definiti all'interno della classe
383 	 * (e naturalmente sue derivate) che abbiano una corrispondenza con un
384 	 * evento e uno dei controlli definiti all'interno di un oggetto wamodulo. Se trova la
385 	 * corrispondenza, associa l'evento al metodo della classe,
386 	 * tale per cui quando l'utente genera l'evento viene invocato il metodo
387 	 * della classe. Inoltre pone lo scope del metodo all'interno della classe
388 	 * derivata da waapplicazione, anzichè di wamodulo/wacontrollo.
389 	 * <p>
390 	 * Ad esempio: supponiamo di avere nella pagina un oggetto wamodulo il cui
391 	 * nome è <b>modulo_ordini</b>, e che per quel modulo vogliate controllare
392 	 * l'evento onblur sul controllo <b>importo</b>
393 	 * </p>
394 	 * Supponiamo ancora che abbiate istanziato in document.wapagina un oggetto
395 	 * di classe derivata da waapplicazione e che implementi il metodo
396 	 * </p>
397 	 * <pre>
398 	 * evento_onblur_modulo_ordini_importo: function (event)
399 	 * </pre>
400 	 * <p>
401 	 * Ebbene, ogni volta che l'utente genererà l'evento verrà invocato il metodo 
402 	 * da voi definito in 
403 	 * document.wapagina, e lo scope del metodo sarà quello di document.wapagina,
404 	 * ossia <b>this</b> corrisponderà a document.wapagina e non a
405 	 * document.modulo_ordini.
406 	 * </p>
407 	 * <p>
408 	 * A questo punto voi potrete, ad esempio, verificare il superamento di una 
409 	 * data soglia:
410 	 * </p>
411 	 * <pre>
412 	 * {
413 	 * if (this.moduli["modulo_ordini"].controlli.importo.dammiValore() > 1000)
414 	 *	alert("Attenzione: superata soglia!");
415 	 * }
416 	 * </pre>
417 	 */
418 	collegaEventiModulo: function ()
419 		{
420 		var elems = [];
421 		var nomeEvento = '';
422 		var nomeControllo = '';
423 		var nomeMetodo = '';
424 		var nomeModulo = '';
425 			
426 		for (nomeMetodo in this)
427 			{
428 			if (typeof(this[nomeMetodo]) == 'function' && nomeMetodo.substr(0, ("evento_").length) == "evento_")
429 				{
430 				elems = nomeMetodo.split("_", 3);
431 				nomeEvento = elems[1];
432 				for (nomeModulo in this.moduli)
433 					{
434 					if (nomeMetodo.substr(0, ("evento_" + nomeEvento + "_" + nomeModulo + "_").length) == 
435 												"evento_" + nomeEvento + "_" + nomeModulo + "_")
436 						{
437 						nomeControllo = nomeMetodo.substr(("evento_" + nomeEvento + "_" + nomeModulo + "_").length);
438 						if (this.moduli[nomeModulo].controlli[nomeControllo])
439 							this.moduli[nomeModulo].controlli[nomeControllo].aggiungiEvento(nomeEvento, this.eventoApplicazione);
440 						else if (nomeModulo == nomeControllo)
441 							// e' stato richiesto un evento sull'intero modulo (verosimilmente l'overload di onsubmit)
442 							this.moduli[nomeModulo].obj[nomeEvento] = this.eventoApplicazione;
443 						}
444 					}
445 				}
446 			}
447 		},

459 	/**
460 	 * apre una pagina figlia
461 	 * 
462 	 * @param {string} pagina URL della pagina di destinazione
463 	 * @param {int} winW larghezza della finestra di destinazione; naturalmente
464 	 * ha senso solo per la modalità di navigazione a finestre; default 1024
465 	 * @param {int} winW altezza della finestra di destinazione; naturalmente
466 	 * ha senso solo per la modalità di navigazione a finestre; default 768
467 	 */
468 	apriPagina: function (pagina, winW, winH)
469 	    {
470 	    if (this.modalitaNavigazione == this.modalitaNavigazioneFinestra)
471 	    	return this.apriPaginaFinestra(pagina, winW, winH);
472 	    else if (this.modalitaNavigazione == this.modalitaNavigazioneInterna)
473 			return this.apriPaginaInterna(pagina);
474 		else
475 		    {
476 		    var qoe = document.location.search.substr(0, 1) == "?" ? "&" : "?";
477 		    var paginaRitorno = encodeBase64(document.location.pathname + 
478 		    					document.location.search + 
479 		    					qoe + 
480 		    					"wa_pry=" + this.dammiSpiazzamentoY());
481 		    qoe = pagina.indexOf("?") != -1 ? "&" : "?";
482 		    location.href = pagina + qoe + "wa_pr=" + paginaRitorno;
483 		    }
484 		
485 	    },

497 								",screenX=0,screenY=0,top=0,left=0" + 
498 								",bgcolor=white,scrollbars=yes,resizable=yes,status=yes");
499 	    },

552 		{
553 		var coppie = (document.location.search.substr(1)).split('&');
554 		for (var i = coppie.length - 1; i >= 0; i--)
555 			{
556 			var kv = coppie[i].split('=');

586 	// di allinearsi alle modifiche da lei effettuate
587 	allineaGenitore: function (valoriInput)
588 	    {
589 	    if (this.modalitaNavigazione == this.modalitaNavigazionePagina)
590 	    	return;
591 	    	
592 		var valoriInput = decodeBase64(valoriInput);
593 		var elems = valoriInput.split("|||");
594 		var param;
595 		valoriInput = {};
596 		for (var i = 0; i < elems.length; i++)
597 			{
598 			 param = elems[i].split("=", 2);
599 			 if (param[0] != "")
600 			 	valoriInput[param[0]] = param[1];
601 			}
602 		var mamma = opener ? opener : parent;
603 		mamma.document.wapagina.metodoAllineamento(valoriInput);
604 	    },

606 	//-------------------------------------------------------------------------
607 	// il default delle funzioni di sincronizzazione che una finestra child 
608 	// chiama per far si che la mamma si allinei; nel caso piu' semplice
609 	// (questo), praticamente si ricarica. se avete bisogno di qualcosa di
610 	// piu' complesso, definite una vostra "metodoAllineamento"
611 	ricaricaPagina: function (datiInputFiglia)
612 		{
613 		if (opener)
614 			{
615 			if (opener.document.wapagina && opener.document.wapagina.metodoAllineamento)
616 				opener.document.wapagina.metodoAllineamento(datiInputFiglia);
617 			}			
618 		else 
619 			parent.document.wapagina.flagRicaricaNonnaNavigazioneInterna = true;
620 			
621 		// in caso di ricaricamento di finestra a causa di inserimento/modifica record
622 		// non dobbiamo chiudere  ne' la finestra di help, ne' quella service, quindi
623 		// dobbiamo ridefinire l'evento onUnload del documento
624 		if (document.layers)
625 			window.captureEvents(Event.ONUNLOAD);
626 		window.onunload = '';
627 		
628 		// ricarichiamo la finestra... sarebbe facile fare una reload; e invece no
629 		// perche' senno' il browser ci chiede l'autorizzazione in caso di precedente
630 		// submit. 
631 		// OCCHIO! si perdono eventuali parametri POST! se servono, fatevi 
632 		// un metodoAllineamento apposta
633 		var qs = '';
634 		if (document.location.search.length)
635 			{
636 			var kv = new Array();
637 			var coppie = (document.location.search.substr(1)).split('&');
638 			for (var i = 0; i < coppie.length; i++)
639 				{
640 				kv = coppie[i].split('=');
641 				if (kv[0] != "wa_pry")
642 					qs += (qs == '' ? '?' : '&') + kv[0] + "=" + (kv[1] ? kv[1] : '');
643 				}
644 			}
645 			
646 		// aggiungiamo al ricaricamnto il parametro per lo spiazzamento verticale,
647 		// cosi' l'utente ritrova la pagina nella posizione dove l'ha lasciata
648 	    if (this.modalitaNavigazione == this.modalitaNavigazioneFinestra)
649 			qs += (qs == '' ? '?' : '&') + "wa_pry=" + this.dammiSpiazzamentoY();
650 	    else if (this.modalitaNavigazione == this.modalitaNavigazioneInterna)
651 			qs += (qs == '' ? '?' : '&') + "wa_pry=" + this.ultimoSpiazzamentoY;
652 		
653 		document.location.href = document.location.pathname + qs;
654 		},
