  4 /**
  5 * classe wacontrollo: contiene attributi e metodi relativi a un controllo
  6 * HTML generico.
  7 * 
  8 * Da questa classe derivano tutti gli altri tipi di controllo del package.
  9 * 
 10 * 
 11 * @class wacontrollo
 12 */
 13 var wacontrollo = new Class
 14 (
 15 	{
 16 	//-------------------------------------------------------------------------
 17 	// proprieta'
 18 	
 19 	/**
 20 	 * oggetto di classe {@link wamodulo} a cui il controllo appartiene
 21 	 * @type wamodulo
 22 	 */
 23 	modulo			: null,
 24 	
 25 	/**
 26 	 * nome del controllo
 27 	 * @type string
 28 	 */
 29 	nome			: '',
 30 	
 31 	/**
 32 	 * tipo del controllo; viene definito da ogni sottoclasse
 33 	 * @type string
 34 	 */
 35 	tipo			: '',
 36 	
 37 	/**
 38 	 * valore del controllo
 39 	 * @type string
 40 	 */
 41 	valore			: '',
 42 	
 43 	/**
 44 	 * indica se il controllo deve essere visibile o meno
 45 	 * @type boolean
 46 	 */
 47 	visibile		: false,
 48 	
 49 	/**
 50 	 * indica se il controllo deve essere readonly o meno
 51 	 * @type boolean
 52 	 */
 53 	solaLettura		: false,
 54 	
 55 	/**
 56 	 * indica se il controllo deve essere obbligatorio o meno
 57 	 * @type boolean
 58 	 */
 59 	obbligatorio	: false,
 60 	
 61 	/**
 62 	 * ogetto HTML contenente il il controllo fisico (questa classe rappresenta
 63 	 * il controllo logico/applicativo; qualche volta i due concetti coincidono,
 64 	 * ma non sempre).
 65 	 * @type HTML_input_object
 66 	 */
 67 	obj				: null,
 68 	
 69 	//-------------------------------------------------------------------------
 70 	/**
 71 	 * inizializzazione (costruttore)
 72 	 * 
 73 	 * @param {wamodulo} modulo valorizza la proprietà {@link modulo}
 74 	 * @param {string} nome valorizza la proprietà {@link nome}
 75 	 * @param {string} valore valorizza la proprietà {@link valore}
 76 	 * @param {boolean} visibile valorizza la proprietà {@link visibile}
 77 	 * @param {boolean} solaLettura valorizza la proprietà {@link solaLettura}
 78 	 * @param {boolean} obbligatorio valorizza la proprietà {@link obbligatorio}
 79 	 */
 80 	initialize: function(modulo, nome, valore, visibile, solaLettura, obbligatorio) 
 81 		{
 82 		// definizione iniziale delle proprieta'
 83 		this.modulo = modulo;
 84 		this.nome = nome;
 85 
 86 		this.valore = valore;
 87 		this.visibile = visibile * 1;
 88 		this.solaLettura = solaLettura * 1;
 89 		this.obbligatorio = obbligatorio * 1;
 90 		this.obj = this.modulo.obj.elements[this.nome];
 91 		if (this.tipo == 'etichetta' || this.tipo == 'cornice')
 92 			this.modulo.etichette[this.nome] = this;
 93 		else
 94 			this.modulo.controlli[this.nome] = this;
 95 		},
 96 		
 97 	//-------------------------------------------------------------------------
 98 	/**
 99 	* inserisce un valore applicativo nel controllo
100 	* 
101 	* @param {mixed} valore valore da inserire nel controllo
102 	*/
103 	mettiValore: function(valore) 
104 		{
105 		this.obj.value = valore;
106 		},
107 		
108 	//-------------------------------------------------------------------------
109 	/**
110 	* restituisce il  valore applicativo contenuto nel controllo
111 	*/
112 	dammiValore: function() 
113 		{
114 		return this.obj.value;
115 		},
116 		
117 	//-------------------------------------------------------------------------
118 	/**
119 	* rende visibile/invisibile un controllo
120 	* 
121 	* @param {boolean} siNo flag di visibilità
122 	* @param {boolean} ancheEtichetta flag che indica che il metodo deve agire 
123 	*	anche sull'eventuale etichetta associata al controllo
124 	*/
125 	mostra: function(siNo, ancheEtichetta) 
126 		{
127 		this.visibile = siNo;
128 		this.renderizza();
129 		if (ancheEtichetta && this.modulo.etichette[this.nome])
130 			this.modulo.etichette[this.nome].mostra(siNo, false);
131 		},
132 		
133 	//-------------------------------------------------------------------------
134 	/**
135 	* abilita/disabilita un controllo
136 	* 
137 	* @param {boolean} siNo flag di abilitazione
138 	* @param {boolean} ancheEtichetta flag che indica che il metodo deve agire 
139 	*	anche sull'eventuale etichetta associata al controllo
140 	*/
141 	abilita: function(siNo, ancheEtichetta) 
142 		{
143 		this.solaLettura = !siNo;
144 		this.renderizza();
145 		if (ancheEtichetta && this.modulo.etichette[this.nome])
146 			this.modulo.etichette[this.nome].abilita(siNo, false);
147 		},
148 		
149 	//-------------------------------------------------------------------------
150 	/**
151 	 * rende obbligatorio/non obbligatorio un controllo
152 	* 
153 	* @param {boolean} siNo flag di obbligatoreità
154 	* @param {boolean} ancheEtichetta flag che indica che il metodo deve agire 
155 	*	anche sull'eventuale etichetta associata al controllo
156 	 */
157 	obbliga: function(siNo, ancheEtichetta) 
158 		{
159 		this.obbligatorio = siNo;
160 		this.renderizza();
161 		if (ancheEtichetta && this.modulo.etichette[this.nome])
162 			this.modulo.etichette[this.nome].obbliga(siNo, false);
163 		},
164 		
165 	//-------------------------------------------------------------------------
166 	/**
167 	* a seconda dello stato /definisce la classe css di un controllo(visibile, 
168 	* solaLettura, obbligatorio) renderizza il controllo
169 	*/
170 	renderizza: function() 
171 		{
172 		this.obj.style.visibility = this.visibile ? '' : 'hidden';
173 		this.obj.disabled = this.solaLettura;
174 		this.obj.className = (this.obbligatorio ? "wamodulo_obbligatorio" : '');
175 		},
176 		
177 	//-------------------------------------------------------------------------
178 	/**
179 	 * verifica che un controllo obbligatorio sia stato valorizzato
180 	 */
181 	verificaObbligo: function() 
182 		{
183 		if (!this.obbligatorio)
184 			return true;
185 		return this.dammiValore() ? true : false;
186 		},
187 		
188 	//-------------------------------------------------------------------------
189 	/**
190 	 * verifica che un controllo sia valorizzato correttamente
191 	 */
192 	verificaForma: function() 
193 		{
194 		return true;
195 		},
196 		
197 	//-------------------------------------------------------------------------
198 	// se la classe ha degli eventi predefiniti, una volta chiamato l'evento
199 	// predefinito occorre innescare anche l'eventuale codice
200 	// definito dal programmatore dell'applicazione per l'evento 
201 	eventoApplicazione: function(event, nomeEvento, objEvento) 
202 		{
203 		if (this[nomeEvento])
204 			{
205 			var obj = objEvento ? objEvento : this.obj;
206 			obj[nomeEvento] = this[nomeEvento];
207 			if (event == undefined)
208 				var esito = obj[nomeEvento]();
209 			else
210 				var esito = obj[nomeEvento](event);
211 			obj[nomeEvento] = this["my" + nomeEvento];
212 			return esito;
213 			}
214 		},
215 		
216 	//-------------------------------------------------------------------------
217 	// associa un evento al controllo; se c'e' un evento predefinito lo
218 	// parcheggia nell'apposita variabile
219 	aggiungiEvento: function (nomeEvento, evento)
220 		{
221 		if (this["my" + nomeEvento])
222 			this[nomeEvento] = evento;
223 		else
224 			this.obj[nomeEvento] = evento;
225 		},
226 	
227 	//-------------------------------------------------------------------------
228 	/**
229 	 * verifica se un controllo fisico appartiene al controllo applicativo
230 	* 
231 	* @param {object} obj oggetto input HTML 
232 	 */
233 	miAppartiene: function (obj)
234 		{
235 		return obj.name == this.obj.name;
236 		},
237 	
238 	//-------------------------------------------------------------------------
239 	/**
240 	 * trim (che non sempre c'e'....)
241 	 */
242 	trim: function(str) 
243 		{
244 		return str.replace(/(^\s*)|(\s*$)/g, "");
245 		},
246 		
247 	//-------------------------------------------------------------------------
248 	// simula un evento sul controllo
249 	simulaEvento: function(tipoEvento) 
250 		{
251 		var event;
252 		if (document.createEvent) 
253 			{
254 			event = document.createEvent("HTMLEvents");
255 			event.initEvent(tipoEvento, true, true);
256 			this.obj.dispatchEvent(event);
257 			} 
258 		else 
259 			{
260 			event = document.createEventObject();
261 			event.eventType = tipoEvento;
262 			this.obj.fireEvent("on" + event.eventType, event);
263 			}
264 		
265 		}
266 		
267 		
268 	}
269 );
