  4 /**
  5  * classe che gestisce le proprietà e i metodi di un modulo waLibs.
  6  * 
  7  * 
  8 * La struttura di questa classe prevede che:
  9 *
 10 * <ul>
 11 * <li>
 12 *	per ogni modulo venga creato un oggetto di classe wamodulo con nome 
 13 * 	uguale a quello dell modulo (proprietà waModulo::nome in PHP) e aggiunto 
 14 * 	come nuova proprietà dell'oggetto document. Questa operazione è svolta
 15 * 	automaticamente dall'XSLT
 16 *	<li>
 17 *	l'XSLT non associa alcun evento ad alcun controllo (se non quello di
 18 *	dei bottoni di submit implicitamente previsto dall'HTML). Sarà compito della
 19 *	applicazione creare le funzioni di gestione degli eventi e associarli ai
 20 *	controlli contenuti nel modulo. La UI di default di waApplicazione 
 21 *	contiene già al suo interno un meccanismo automatico per fare ciò.
 22 * </ul>
 23 * 
 24 * @class wamodulo
 25  */
 26 var wamodulo = new Class
 27 (
 28 	{
 29 	//-------------------------------------------------------------------------
 30 	// proprieta'
 31 	
 32 	/**
 33 	 * nome del modulo
 34 	 * @type string
 35 	 */
 36 	nome				: '',
 37 	
 38 	/**
 39 	 * flag deve essere valorizzato dai bottoni di submit quando si desidera 
 40 	 * eliminare un record (bottoni con proprietà {@link wabottone#elimina} = 
 41 	 * true); viene invocato il submit ma non vengono eseguiti 
 42 	 * controlli formali o di obbligatorietà
 43 	 * @type boolean
 44 	 */
 45 	tipoInvioElimina	: false,	
 46 	
 47 	/**
 48 	 * flag deve essere valorizzato dai bottoni quando si desidera 
 49 	 * abortire la sessione di editing (bottoni con proprietà {@link wabottone#annulla} = 
 50 	 * true); non vengono eseguiti controlli formali o di obbligatorietà
 51 	 * @type boolean
 52 	 */
 53 	tipoInvioAnnulla	: false,	// deve essere valorizzato dai bottoni di submit se si desidera abortire l'operazione e non consolidare nulla; non vengono eseguiti controlli formali o di obbligatorieta'
 54 
 55 	/**
 56 	 * dizionario (array associativo; in javascript è un oggetto, non un array)
 57 	 * dei controlli di input (no etichette, no cornici) contenuti nel modulo.
 58 	 * 
 59 	 * La chiave di ogni elemento è il nome del controllo.
 60 	 * 
 61 	 * @type object
 62 	 */
 63 	controlli			: {},
 64 
 65 	/**
 66 	 * dizionario (array associativo; in javascript è un oggetto, non un array)
 67 	 * dei controlli di tipo etichetta o cornice contenuti nel modulo.
 68 	 * 
 69 	 * La chiave di ogni elemento è il nome dell'etichetta.
 70 	 * 
 71 	 * @type object
 72 	 */
 73 	etichette			: {},
 74 
 75 	/**
 76 	 * oggetto form HTML fisico a cui fa riferimento il modulo
 77 	 * 
 78 	 * @type object
 79 	 */
 80 	obj					: null,
 81 	
 82 	/**
 83 	 * se il flag è alzato, allora viene mostrato il contenutoi del messaggio
 84 	 * RPC ricevuto dal server
 85 	 * 
 86 	 * @type boolean
 87 	 */
 88 	debugRPC					: false,
 89 	
 90 //	applicazione		: null,		// eventuale istanza dell'applicazione (e' lei stessa a valorizzare questa proprieta')
 91 	http_request 		: false,
 92 	errore_rpc 			: 'wamodulo_errore_rpc',
 93 	
 94 	//-------------------------------------------------------------------------
 95 	// implements
 96 
 97 	//-------------------------------------------------------------------------
 98 	/**
 99 	 * inizializzazione (costruttore)
100 	 * 
101 	 * @param {string} nome valorizza la proprietà {@link nome}
102 	 */
103 	initialize: function(nome) 
104 		{
105 		this.nome = nome;
106 		this.obj = document.getElementById(this.nome);
107 		this.obj.wamodulo = this;
108 		this.obj.onsubmit = function onsubmit(event) {return this.wamodulo.validaModulo();}
109 		},
110 		
111 	//-------------------------------------------------------------------------
112 	/**
113 	 *  verifica che tutti i controlli obbligatori siano stati valorizzati
114 	 */
115 	verificaObbligo: function() 
116 		{
117 		msg = '';
118 		for (nomeControllo in this.controlli)
119 			{
120 			if (!this.controlli[nomeControllo].verificaObbligo())
121 				msg += "Il campo " + 
122 						(this.etichette[nomeControllo] ? this.etichette[nomeControllo].dammiValore() : nomeControllo) +
123 						" e' obbligatorio e non e' stato valorizzato.\n";
124 			}
125 			
126 		if (msg == '')
127 			return true;
128 
129 		alert(msg);
130 		return false;
131 		},
132 		
133 	//-------------------------------------------------------------------------
134 	/**
135 	 * verifica che tutti i controlli siano stati valorizzati con valori
136 	 * compatibili col tipo del controllo
137 	 */
138 	verificaForma: function() 
139 		{
140 		msg = '';
141 		for (nomeControllo in this.controlli)
142 			{
143 			if (!this.controlli[nomeControllo].verificaForma())
144 				msg += "Il campo " + 
145 						(this.etichette[nomeControllo] ? this.etichette[nomeControllo].dammiValore() : nomeControllo) +
146 						" non e' stato valorizzato correttamente.\n";
147 			}
148 			
149 		if (msg == '')
150 			return true;
151 
152 		alert(msg);
153 		return false;
154 		},
155 		
156 	//-------------------------------------------------------------------------
157 	/**
158 	 * funzione di validazione standard di un modulo
159 	 */
160 	validaModulo: function() 
161 		{
162 		if (this.tipoInvioAnnulla)
163 			{
164 			this.obj.wamodulo_operazione.value = 5;
165 			return true;
166 			}
167 		if (this.tipoInvioElimina)
168 			{
169 			this.tipoInvioElimina = false;
170 			if (confirm('Confermi eliminazione?'))
171 				{
172 				this.obj.wamodulo_operazione.value = 4;
173 				return true;
174 				}
175 			return false;
176 			}
177 		if (this.verificaForma())
178 			{
179 			if(this.verificaObbligo())
180 				return confirm('Confermi operazione?');
181 			}
182 		return false;
183 		},
184 		
185 	//-------------------------------------------------------------------------
186 	//-------------------------------------------------------------------------
187 	//-------------------------------------------------------------------------
188 	// restituisce una connessione rpc
189 	getHttpRequestObj: function() 
190 		{
191 		var hr = false;
192 		if (window.ActiveXObject) 
193 			{ 
194 			// IE
195 			try 
196 				{
197 				hr = new ActiveXObject("Msxml2.XMLHTTP");
198 				}
199 			catch (e) 
200 				{
201 				try 
202 					{
203 					hr = new ActiveXObject("Microsoft.XMLHTTP");
204 					}
205 				catch (e) 
206 					{}
207 				}			
208 			}
209 		else if (window.XMLHttpRequest) 
210 			{ 
211 			// Mozilla, Safari,...
212 			hr = new XMLHttpRequest();
213 //			if (hr.overrideMimeType) 
214 //				hr.overrideMimeType('text/plain');
215 			}
216 			
217 		
218 		return hr;
219 		},
220 		
221 	//-------------------------------------------------------------------------
222 	// mostra un eventuale errore durante rpc
223 	mostraErroreRPC: function(msg) 
224 		{
225 		alert(msg);
226 		return this.errore_rpc;
227 		},
228 		
229 	//-------------------------------------------------------------------------
230 	// cerca di leggere dalla struttura xml-rpc un valore
231 	leggiValoreXML: function(xmlDoc, chiave) 
232 		{
233 			
234 		var retval = '';
235 		
236 		if (xmlDoc.getElementsByTagName(chiave) && 
237 			xmlDoc.getElementsByTagName(chiave)[0] && 
238 			xmlDoc.getElementsByTagName(chiave)[0].firstChild)
239 			return xmlDoc.getElementsByTagName(chiave)[0].firstChild.nodeValue;
240 
241 		return "";
242 
243 		},
244 	
245 	//-------------------------------------------------------------------------
246 	/**
247 	 * effettua una chiamata RPC verso il server.
248 	 * 
249 	* gli argomenti da passare al metodo sono posizionali 
250 	* <ol>
251 	* <li>nome funzione/metodo PHP da richiamare lato server
252 	* <li>parametro 1 da passare alla funzione/metodo
253 	* <li>parametro 2 da passare alla funzione/metodo
254 	* <li>parametro n da passare alla funzione/metodo...
255 	* </ol>
256 	* 
257 	 */
258 	RPC: function() 
259 		{
260 		
261 		try
262 			{
263 			var http_request = this.getHttpRequestObj();
264 			
265 			// creiamo il messaggio da inviare (parametri della chiamata)
266 			// 6 e' il codice operazione rpc (vedi defines in wamodulo.inc.php)
267 			var post = 'wamodulo_operazione=6';
268 			// passiamo il nome del modulo che richiede RPC, in modo da
269 			// riconoscere, lato server, quale modulo ha richiesto l'operazione
270 			post += '&wamodulo_nome_modulo=' + escape(this.nome);
271 			// passiamo in wamodulo_funzionerpc il nome della funzione/metodo 
272 			// PHP da richiamare, che deve essere il primo argomento con cui e'
273 			// stato chiamato il presente metodo
274 			post += '&wamodulo_funzione_rpc=' + escape(arguments[0]);
275 			// passiamo i parametri coi quali il presente metodo e' stato 
276 			// richiamato nel medesimo ordine; li ricevera' nel medesimo ordine
277 			// posizionale anche la funzione/metodo PHP lato server
278 			for (var i = 1; i < arguments.length; i++)
279 					post += '&wamodulo_dati_rpc[' + (i - 1) + ']=' + escape(arguments[i]);
280 
281 			// chiamiamo il server
282 			http_request.open('POST', document.location.href, false);
283 			http_request.setRequestHeader("Content-Type", "application/x-www-form-urlencoded; charset=UTF-8");
284 			http_request.send(post);
285 			
286 			// verifica eventuali errori di sistema
287 			if (http_request.status != 200) 
288 				return this.mostraErroreRPC("Errore durante RPC: " + http_request.status);
289 				
290 			// gestione della risposta
291 			if (this.debugRPC)
292 				alert(http_request.responseText);
293 			var xmlDoc = http_request.responseXML;				
294 			var esito = this.leggiValoreXML(xmlDoc, 'wamodulo_esito_rpc');
295 			var messaggio = this.leggiValoreXML(xmlDoc, 'wamodulo_messaggio_rpc');
296 			if (esito != '__rpc_ok__')
297 				return this.mostraErroreRPC("Errore applicativo server durante chiamata RPC:\n" + messaggio);
298 			
299 			if (xmlDoc.getElementsByTagName('item')[0])
300 				{
301 				// l'esito e' un array; torniamo UN OGGETTO (dizionario!) perche' le 
302 				// chiavi non e' detto che siano necessariamente numeriche
303 				var item = null;
304 				var retval = {};
305 				if (xmlDoc.getElementsByTagName('item')[0].getAttribute("id"))
306 					{
307 					// l'array non e' vuoto	
308 					for (var i = 0;  i < xmlDoc.getElementsByTagName('item').length; i++)
309 						{
310 						item = xmlDoc.getElementsByTagName('item')[i];
311 						retval[item.getAttribute("id")] = item.childNodes[0] ? item.childNodes[0].nodeValue : '';
312 						}
313 					}					
314 				}
315 			else 
316 				var retval = this.leggiValoreXML(xmlDoc, 'wamodulo_dati_rpc');
317 						
318 			return retval;
319 			}
320 		catch(exception)
321 			{
322 			return this.mostraErroreRPC("Errore durante RPC: " + exception);
323 			}
324 		}
325 		
326 	}
327 );
