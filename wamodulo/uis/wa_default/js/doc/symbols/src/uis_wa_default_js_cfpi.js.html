  4 /**
  5 * classe wacfpi: controllo di input di un codice fiscale e/o partita IVA
  6 * 
  7 * @class wacfpi
  8 * @extends wacontrollo
  9 */
 10 var wacfpi = new Class
 11 (
 12 	{
 13 	//-------------------------------------------------------------------------
 14 	// extends
 15 	Extends: wacontrollo,
 16 
 17 	//-------------------------------------------------------------------------
 18 	// proprieta'
 19 	tipo: 'cfpi',
 20 	gestioneCF: true,
 21 	gestionePI: false,
 22 		
 23 	//-------------------------------------------------------------------------
 24 	//initialization
 25 	initialize: function(modulo, nome, valore, visibile, solaLettura, obbligatorio) 
 26 		{
 27 		// definizione iniziale delle proprieta'
 28 		this.parent(modulo, nome, valore, visibile, solaLettura, obbligatorio);
 29 		this.gestioneCF = this.obj.gestioneCF == '1';
 30 		this.gestionePI = this.obj.gestionePI == '1';
 31 		
 32 		},
 33 		
 34 	//-------------------------------------------------------------------------
 35 	verificaForma: function() 
 36 		{
 37 		if (this.obj.value == '')
 38 			return true;
 39 		
 40 		var tester = this.obj.value.substring(0, 1).toUpperCase();
 41 		if ((/^[A-Z]$/).test(tester) && this.gestioneCF)
 42 			return this.verificaCF();
 43 		else if ((/^\d$/).test(tester) && this.gestionePI)
 44 			return this.verificaPI();
 45 
 46 		return false;
 47 		},
 48 		
 49 	//-------------------------------------------------------------------------
 50 	verificaCF: function() 
 51 		{
 52 		var cf = this.obj.value.toUpperCase();
 53 		var set1 = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
 54 		var set2 = "ABCDEFGHIJABCDEFGHIJKLMNOPQRSTUVWXYZ";
 55 		var setpari = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
 56 		var setdisp = "BAKPLCQDREVOSFTGUHMINJWZYX";
 57 		var s = 0;
 58 		for( i = 1; i <= 13; i += 2 )
 59 		   s += setpari.indexOf( set2.charAt( set1.indexOf( cf.charAt(i) )));
 60 		for( i = 0; i <= 14; i += 2 )
 61 		   s += setdisp.indexOf( set2.charAt( set1.indexOf( cf.charAt(i) )));
 62 		if ( s%26 != cf.charCodeAt(15)-'A'.charCodeAt(0) )
 63 			{
 64 			// il check digit è sbagliato...
 65 //			alert(String.fromCharCode(s % 26 + 65));
 66 			return false;
 67 			}
 68 		
 69 		var cfReg = /^[A-Z]{6}\d{2}[A-EHLMPRST]\d{2}[A-Z]\d{3}[A-Z]$/;
 70 		if (cfReg.test(cf))
 71 			// codice standard, non c'e' bisogno di controllare omocodia
 72 			return true;
 73 
 74 		// verifichiamo che la parte letterale sia composta effettivamente di
 75 		// lettere e che la parte normalmente numerica contenga solo lettere
 76 		// previste nei casi di omocodia
 77 		cfReg = /^[A-Z]{6}[\dL-NP-V]{2}[A-EHLMPRST][\dL-NP-V]{2}[A-Z][\dL-NP-V]{3}[A-Z]$/;
 78 		if (!cfReg.test(cf))
 79 			return false;
 80 
 81 		// una eventuale lettera al posto del numero indicante la decina del mese
 82 		// non può essere una U o una V (corrispondenti a 8 e 9)
 83 		cfReg = /^[\dA-Z]{9}[\dL-NP-T][\dA-Z]{6}$/;
 84 		if (!cfReg.test(cf))
 85 			return false;
 86 
 87 		return true;
 88 		},
 89 		
 90 	//-------------------------------------------------------------------------
 91 	verificaPI: function() 
 92 		{
 93 		var piva = this.obj.value;
 94 
 95 		var piReg = /^\d{11}$/;
 96 		if (!piReg.test(piva))
 97 			// composizione sbagliata
 98 			return false;
 99 
100         var X = 0 ;
101         var Y = 0 ;
102         var Z = 0 ;
103     
104         // cifre posto dispari ... ma per un array indicizzato a zero, la prima cifra ha indice zero ... appunto !
105         X += parseInt( piva.charAt(0) ) ;
106         X += parseInt( piva.charAt(2) ) ;
107         X += parseInt( piva.charAt(4) ) ;
108         X += parseInt( piva.charAt(6) ) ;
109         X += parseInt( piva.charAt(8) ) ;
110 
111         // cifre posto pari ... ma per un array indicizzato a zero, la prima cifra ha indice uno ...
112         Y += 2 * parseInt( piva.charAt(1) ) ;    if ( parseInt( piva.charAt(1) ) >= 5 ) Z++ ;
113         Y += 2 * parseInt( piva.charAt(3) ) ;    if ( parseInt( piva.charAt(3) ) >= 5 ) Z++ ;
114         Y += 2 * parseInt( piva.charAt(5) ) ;    if ( parseInt( piva.charAt(5) ) >= 5 ) Z++ ;
115         Y += 2 * parseInt( piva.charAt(7) ) ;    if ( parseInt( piva.charAt(7) ) >= 5 ) Z++ ;
116         Y += 2 * parseInt( piva.charAt(9) ) ;    if ( parseInt( piva.charAt(9) ) >= 5 ) Z++ ;
117         
118         var T = ( X + Y + Z ) % 10 ;
119         var C = ( 10 - T ) % 10 ;
120 
121         return ( piva.charAt( piva.length - 1 ) == C );
122 		}
123 		
124 	}
125 );
