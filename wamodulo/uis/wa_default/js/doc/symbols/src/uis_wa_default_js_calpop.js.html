 82 function Calendar_get_daysofmonth(monthNo, p_year) {
 83 	/* 
 84 	Check for leap year ..
 85 	1.Years evenly divisible by four are normally leap years, except for... 
 86 	2.Years also evenly divisible by 100 are not leap years, except for... 
 87 	3.Years also evenly divisible by 400 are leap years. 
 88 	*/
 89 	if ((p_year % 4) == 0) {
 90 		if ((p_year % 100) == 0 && (p_year % 400) != 0)
 91 			return Calendar.DOMonth[monthNo];
 92 	
 93 		return Calendar.lDOMonth[monthNo];
 94 	} else
 95 		return Calendar.DOMonth[monthNo];
 96 }

 98 function Calendar_calc_month_year(p_Month, p_Year, incr) {
 99 	/* 
100 	Will return an 1-D array with 1st element being the calculated month 
101 	and second being the calculated year 
102 	after applying the month increment/decrement as specified by 'incr' parameter.
103 	'incr' will normally have 1/-1 to navigate thru the months.
104 	*/
105 	var ret_arr = new Array();
106 	
107 	if (incr == -1) {
108 		// B A C K W A R D
109 		if (p_Month == 0) {
110 			ret_arr[0] = 11;
111 			ret_arr[1] = parseInt(p_Year) - 1;
112 		}
113 		else {
114 			ret_arr[0] = parseInt(p_Month) - 1;
115 			ret_arr[1] = parseInt(p_Year);
116 		}
117 	} else if (incr == 1) {
118 		// F O R W A R D
119 		if (p_Month == 11) {
120 			ret_arr[0] = 0;
121 			ret_arr[1] = parseInt(p_Year) + 1;
122 		}
123 		else {
124 			ret_arr[0] = parseInt(p_Month) + 1;
125 			ret_arr[1] = parseInt(p_Year);
126 		}
127 	}
128 	
129 	return ret_arr;
130 }

308 Calendar.prototype.cal_data = function() {
309 	var vDate = new Date();
310 	vDate.setDate(1);
311 	vDate.setMonth(this.gMonth);
312 	vDate.setFullYear(this.gYear);
313 
314 	var vFirstDay=vDate.getDay();
315 	var vDay=1;
316 	var vLastDay=Calendar.get_daysofmonth(this.gMonth, this.gYear);
317 	var vOnLastDay=0;
318 	var vCode = "";
319 
320 	/*
321 	Get day for the 1st of the requested month/year..
322 	Place as many blank cells before the 1st day of the month as necessary. 
323 	*/
324 	vCode = vCode + "<TR>";
325 	for (i=0; i<vFirstDay; i++) {
326 		vCode = vCode + "<TD WIDTH='14%'" + this.write_weekend_string(i) + "><FONT FACE='" + fontface + "'> </FONT></TD>";
327 	}
328 
329 	// Write rest of the 1st week
330 	for (j=vFirstDay; j<7; j++) {
331 		vCode = vCode + "<TD WIDTH='14%'" + this.write_weekend_string(j) + "><FONT FACE='" + fontface + "'>" + 
332 			"<A HREF='javascript:void(0);' " + 
333 				"onMouseOver=\"window.status='set date to " + this.format_data(vDay) + "'; return true;\" " +
334 				"onMouseOut=\"window.status=' '; return true;\" " +
335 				"onClick=\"document." + this.gReturnItem + ".value='" +
336 				this.format_data(vDay) +
337 				// inizio modificato bicio
338 				// "';ggPosX=-1;ggPosY=-1;nd();nd();\">" +
339 				"';updateDateCB('" + this.gReturnItem + "');ggPosX=-1;ggPosY=-1;nd();nd();\" class=\"lcalpop\">" +
340 				// fine modificato bicio
341 				this.format_day(vDay) +
342 			"</A>" +
343 			"</FONT></TD>";
344 		vDay=vDay + 1;
345 	}
346 	vCode = vCode + "</TR>";
347 
348 	// Write the rest of the weeks
349 	for (k=2; k<7; k++) {
350 		vCode = vCode + "<TR>";
351 
352 		for (j=0; j<7; j++) {
353 			vCode = vCode + "<TD WIDTH='14%'" + this.write_weekend_string(j) + "><FONT FACE='" + fontface + "'>" +
354 				"<A HREF='javascript:void(0);' " +
355 					"onMouseOver=\"window.status='set date to " + this.format_data(vDay) + "'; return true;\" " +
356 					"onMouseOut=\"window.status=' '; return true;\" " +
357 
358 					"onClick=\"document." + this.gReturnItem + ".value='" +
359 					this.format_data(vDay) +
360 					// inizio modificato bicio
361 					// "';window.scroll(0,ggPosY);ggPosX=-1;ggPosY=-1;nd();nd();\">" +
362 					"';updateDateCB('" + this.gReturnItem + "');window.scroll(0,ggPosY);ggPosX=-1;ggPosY=-1;nd();nd();\" class=\"lcalpop\">" +
363 					// fine modificato bicio
364 
365 				this.format_day(vDay) +
366 				"</A>" + 
367 				"</FONT></TD>";
368 			vDay=vDay + 1;
369 
370 			if (vDay > vLastDay) {
371 				vOnLastDay = 1;
372 				break;
373 			}
374 		}
375 
376 		if (j == 6)
377 			vCode = vCode + "</TR>";
378 		if (vOnLastDay == 1)
379 			break;
380 	}
381 	
382 	// Fill up the rest of last week with proper blanks, so that we get proper square blocks
383 	for (m=1; m<(7-j); m++) {
384 		if (this.gYearly)
385 			vCode = vCode + "<TD WIDTH='14%'" + this.write_weekend_string(j+m) + 
386 			"><FONT FACE='" + fontface + "' COLOR='gray'> </FONT></TD>";
387 		else
388 			vCode = vCode + "<TD WIDTH='14%'" + this.write_weekend_string(j+m) + 
389 			"><FONT FACE='" + fontface + "' COLOR='gray'>" + m + "</FONT></TD>";
390 	}
391 	
392 	return vCode;
393 }

395 Calendar.prototype.format_day = function(vday) {
396 	var vNowDay = gNow.getDate();
397 	var vNowMonth = gNow.getMonth();
398 	var vNowYear = gNow.getFullYear();
399 
400 	if (vday == vNowDay && this.gMonth == vNowMonth && this.gYear == vNowYear)
401 		return ("<FONT COLOR=\"RED\"><B>" + vday + "</B></FONT>");
402 	else
403 		return (vday);
404 }

411 		if (vday == weekend[i])
412 			return (" BGCOLOR=\"" + weekendColor + "\"");
413 	}
414 	
415 	return "";

418 Calendar.prototype.format_data = function(p_day) {
419 	var vData;
420 	var vMonth = 1 + this.gMonth;
421 	vMonth = (vMonth.toString().length < 2) ? "0" + vMonth : vMonth;
422 	var vMon = Calendar.get_month(this.gMonth).substr(0,3).toUpperCase();
423 	var vFMon = Calendar.get_month(this.gMonth).toUpperCase();
424 	var vY4 = new String(this.gYear);
425 	var vY2 = new String(this.gYear.substr(2,2));
426 	var vDD = (p_day.toString().length < 2) ? "0" + p_day : p_day;
427 
428 	switch (this.gFormat) {
429 		case "MM\/DD\/YYYY" :
430 			vData = vMonth + "\/" + vDD + "\/" + vY4;
431 			break;
432 		case "MM\/DD\/YY" :
433 			vData = vMonth + "\/" + vDD + "\/" + vY2;
434 			break;
435 		case "MM-DD-YYYY" :
436 			vData = vMonth + "-" + vDD + "-" + vY4;
437 			break;
438 		case "YYYY-MM-DD" :
439 			vData = vY4 + "-" + vMonth + "-" + vDD;
440 			break;
441 		case "MM-DD-YY" :
442 			vData = vMonth + "-" + vDD + "-" + vY2;
443 			break;
444 		case "DD\/MON\/YYYY" :
445 			vData = vDD + "\/" + vMon + "\/" + vY4;
446 			break;
447 		case "DD\/MON\/YY" :
448 			vData = vDD + "\/" + vMon + "\/" + vY2;
449 			break;
450 		case "DD-MON-YYYY" :
451 			vData = vDD + "-" + vMon + "-" + vY4;
452 			break;
453 		case "DD-MON-YY" :
454 			vData = vDD + "-" + vMon + "-" + vY2;
455 			break;
456 		case "DD\/MONTH\/YYYY" :
457 			vData = vDD + "\/" + vFMon + "\/" + vY4;
458 			break;
459 		case "DD\/MONTH\/YY" :
460 			vData = vDD + "\/" + vFMon + "\/" + vY2;
461 			break;
462 		case "DD-MONTH-YYYY" :
463 			vData = vDD + "-" + vFMon + "-" + vY4;
464 			break;
465 		case "DD-MONTH-YY" :
466 			vData = vDD + "-" + vFMon + "-" + vY2;
467 			break;
468 		case "DD\/MM\/YYYY" :
469 			vData = vDD + "\/" + vMonth + "\/" + vY4;
470 			break;
471 		case "DD\/MM\/YY" :
472 			vData = vDD + "\/" + vMonth + "\/" + vY2;
473 			break;
474 		case "DD-MM-YYYY" :
475 			vData = vDD + "-" + vMonth + "-" + vY4;
476 			break;
477 		case "DD-MM-YY" :
478 			vData = vDD + "-" + vMonth + "-" + vY2;
479 			break;
480 		default :
481 			vData = vMonth + "\/" + vDD + "\/" + vY4;
482 	}
483 
484 	return vData;
485 }

487 function Build(p_item, p_month, p_year, p_format) {
488 	gCal = new Calendar(p_item, p_month, p_year, p_format);
489 
490 	// Customize your Calendar here..
491 	gCal.gBGColor="white";
492 	gCal.gLinkColor="black";
493 	gCal.gTextColor="black";
494 	gCal.gHeaderColor="darkgreen";
495 
496 	// initialize the content string
497 	ggWinContent = "";
498 
499 	// Choose appropriate show function
500 	if (gCal.gYearly) {
501 		// and, since the yearly calendar is so large, override the positioning and fontsize
502 		// warning: in IE6, it appears that "select" fields on the form will still show
503 		//	through the "over" div; Note: you can set these variables as part of the onClick
504 		//	javascript code before you call the show_yearly_calendar function
505 		if (ggPosX == -1) ggPosX = 10;
506 		if (ggPosY == -1) ggPosY = 10;
507 		if (fontsize == 8) fontsize = 6;
508 		// generate the calendar
509 		gCal.showY();
510 		}
511 	else {
512 		gCal.show();
513 		}
514 
515 	// if this is the first calendar popup, use autopositioning with an offset
516 	if (ggPosX == -1 && ggPosY == -1) {
517 		overlib(ggWinContent, AUTOSTATUSCAP, STICKY, CLOSECLICK, CSSSTYLE,
518 			TEXTSIZEUNIT, "pt", TEXTSIZE, 8, CAPTIONSIZEUNIT, "pt", CAPTIONSIZE, 8, CLOSESIZEUNIT, "pt", CLOSESIZE, 8,
519 			CAPTION, "Seleziona un giorno", OFFSETX, 20, OFFSETY, -20);
520 		// save where the 'over' div ended up; we want to stay in the same place if the user
521 		//	clicks on one of the year or month navigation links
522 		if ( (ns4) || (ie4) ) {
523 		        ggPosX = parseInt(over.left);
524 		        ggPosY = parseInt(over.top);
525 			} else if (ns6) {
526 			ggPosX = parseInt(over.style.left);
527 			ggPosY = parseInt(over.style.top);
528 			}
529 		}
530 	else {
531 		// we have a saved X & Y position, so use those with the FIXX and FIXY options
532 		overlib(ggWinContent, AUTOSTATUSCAP, STICKY, CLOSECLICK, CSSSTYLE,

534 			CAPTION, "Seleziona un giorno", FIXX, ggPosX, FIXY, ggPosY);
535 		}
536 	window.scroll(ggPosX, ggPosY);
537 }
