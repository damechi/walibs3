1 //*****************************************************************************
  2 //*****************************************************************************
  3 //*****************************************************************************
  4 /**
  5 * classe waora: controllo composito per l'input di una ora
  6 * 
  7 * @class waora
  8 * @extends wacontrollo
  9 */
 10 var waora = new Class
 11 (
 12 	{
 13 	//-------------------------------------------------------------------------
 14 	// extends
 15 	Extends: wacontrollo,
 16 
 17 	//-------------------------------------------------------------------------
 18 	// proprieta'
 19 	tipo: 'ora',
 20 	
 21 	/**
 22 	 * oggetto HTML fisico utilizzato per selezionare l'ora
 23 	 * @type HTML_object
 24 	 */
 25 	selezioneOra: null,
 26 	
 27 	/**
 28 	 * oggetto HTML fisico utilizzato per selezionare i minuti
 29 	 * @type HTML_object
 30 	 */
 31 	selezioneMin: null,
 32 	
 33 	/**
 34 	 * oggetto HTML fisico utilizzato per selezionare i secondi
 35 	 * @type HTML_object
 36 	 */
 37 	selezioneSec: null,
 38 	
 39 	//-------------------------------------------------------------------------
 40 	//initialization
 41 	initialize: function(modulo, nome, valore, visibile, solaLettura, obbligatorio) 
 42 		{
 43 		// definizione iniziale delle proprieta'
 44 		this.parent(modulo, nome, valore, visibile, solaLettura, obbligatorio);
 45 		this.selezioneOra = this.modulo.obj.elements["wamodulo_ora_" + this.nome];
 46 		this.selezioneMin = this.modulo.obj.elements["wamodulo_min_" + this.nome];
 47 		this.selezioneSec = this.modulo.obj.elements["wamodulo_sec_" + this.nome];
 48 		},
 49 		
 50 	//-------------------------------------------------------------------------
 51 	// restituisce il valore logico del controllo
 52 	dammiValore: function() 
 53 		{
 54 		if (this.selezioneOra.value == '' || this.selezioneMin.value == '' ||
 55 			(this.selezioneSec && this.selezioneSec.value == ''))
 56 			return false;
 57 //		return new Date(1980, 0, 1,
 58 //						this.selezioneOra.value, this.selezioneMin.value, this.selezioneSec ? this.selezioneSec.value : 0);
 59 		var retval = new Date();
 60 		retval.setUTCFullYear(1980, 0, 1);
 61 		retval.setUTCHours(this.selezioneOra.value, this.selezioneMin.value, this.selezioneSec ? this.selezioneSec.value : 0);
 62 		return retval;
 63 		},
 64 		
 65 	//-------------------------------------------------------------------------
 66 	// inserisce un valore logico nel controllo
 67 	mettiValore: function(valore) 
 68 		{
 69 		if (!valore || valore == 0 || valore == '')
 70 			{
 71 			this.selezioneOra.value = this.selezioneMin.value = '';
 72 			if (this.selezioneSec)
 73 				this.selezioneSec.value = '';
 74 			}
 75 		else
 76 			{
 77 			this.selezioneOra.value = valore.getUTCHours(); 
 78 			this.selezioneMin.value = valore.getUTCMinutes(); 
 79 			if (this.selezioneSec)
 80 				this.selezioneSec.value = valore.getUTCSeconds();
 81 			}
 82 		},
 83 		
 84 	//-------------------------------------------------------------------------
 85 	verificaForma: function() 
 86 		{
 87 		if (!this.dammiValore())
 88 			return true;
 89 			
 90 //		var testDate = new Date(1980, 0, 1,
 91 //						this.selezioneOra.value, this.selezioneMin.value, this.selezioneSec ? this.selezioneSec.value : 0);
 92 		var testDate = new Date();
 93 		testDate.setUTCFullYear(1980, 0, 1);
 94 		testDate.setUTCHours(this.selezioneOra.value, this.selezioneMin.value, this.selezioneSec ? this.selezioneSec.value : 0);
 95 		return this.selezioneOra.value == testDate.getUTCHours() &&
 96 					this.selezioneMin.value == testDate.getUTCMinutes() &&
 97 					(this.selezioneSec ? this.selezioneSec.value == testDate.getUTCSeconds() : true);
 98 		},
 99 		
100 	//-------------------------------------------------------------------------
101 	// a seconda dello stato definisce la classe css di un controllo
102 	renderizza: function() 
103 		{
104 		this.selezioneOra.style.visibility = this.selezioneMin.style.visibility = this.visibile ? '' : 'hidden';
105 		if (this.selezioneSec)
106 			this.selezioneSec.style.visibility = this.selezioneOra.style.visibility;
107 			
108 		this.selezioneOra.disabled = this.selezioneMin.disabled = this.solaLettura;
109 		if (this.selezioneSec)
110 			this.selezioneSec.disabled = this.selezioneOra.disabled;
111 		
112 		this.selezioneOra.className = this.selezioneMin.className = 
113 								(this.obbligatorio ? "wamodulo_obbligatorio" : '');
114 								
115 		if (this.selezioneSec)
116 			this.selezioneSec.className = this.selezioneOra.className;
117 		},
118 		
119 	//-------------------------------------------------------------------------
120 	// associa un evento al controllo
121 	aggiungiEvento: function (nomeEvento, evento)
122 		{
123 		this.selezioneOra[nomeEvento] = evento;
124 		this.selezioneMin[nomeEvento] = evento;
125 		if (this.selezioneSec)
126 			this.selezioneSec[nomeEvento] = evento;
127 		},
128 	
129 	//-------------------------------------------------------------------------
130 	// verifica se un controllo fisico appartiene al controllo logico
131 	miAppartiene: function (obj)
132 		{
133 		if  (obj.name == this.selezioneOra.name || obj.name == this.selezioneMin.name)
134 			return true;
135 		if (this.selezioneSec)
136 			return obj.name == this.selezioneSec.name;
137 		return false;
138 		},
139 	
140 	//-------------------------------------------------------------------------
141 	// simula un evento sul controllo
142 	simulaEvento: function(tipoEvento) 
143 		{
144 		var event;
145 		if (document.createEvent) 
146 			{
147 			event = document.createEvent("HTMLEvents");
148 			event.initEvent(tipoEvento, true, true);
149 			this.selezioneOra.dispatchEvent(event);
150 			} 
151 		else 
152 			{
153 			event = document.createEventObject();
154 			event.eventType = tipoEvento;
155 			this.selezioneOra.fireEvent("on" + event.eventType, event);
156 			}
157 		
158 		}
159 		
160 	}
161 );
