  4 /**
  5 * classe waetichetta: label HTML tipicamente associata ad un controllo di
  6 * input col medesimo nome
  7 * 
  8 * @class waetichetta
  9 * @extends wacontrollo
 10 */
 11 var waetichetta = new Class
 12 (
 13 	{
 14 	//-------------------------------------------------------------------------
 15 	// extends
 16 	Extends: wacontrollo,
 17 
 18 	//-------------------------------------------------------------------------
 19 	// proprieta'
 20 	tipo: 'etichetta',
 21 	helpLink: false,
 22 	
 23 	//-------------------------------------------------------------------------
 24 	//initialization
 25 	initialize: function(modulo, nome, valore, visibile, solaLettura, obbligatorio) 
 26 		{
 27 		// definizione iniziale delle proprieta'
 28 		this.parent(modulo, nome, valore, visibile, solaLettura, obbligatorio);
 29 		this.obj = document.getElementById(this.modulo.nome + "_"+ this.nome);
 30 		this.helpLink = document.getElementById("hlplink_" + this.modulo.nome + "_"+ this.nome);
 31 		},
 32 		
 33 	//-------------------------------------------------------------------------
 34 	// inserisce un valore logico nel controllo
 35 	mettiValore: function(valore) 
 36 		{
 37 		if (this.helpLink)
 38 			this.helpLink.innerHTML = valore;
 39 		else
 40 			this.obj.innerHTML = valore;
 41 		},
 42 		
 43 	//-------------------------------------------------------------------------
 44 	// restituisce il valore logico del controllo
 45 	dammiValore: function() 
 46 		{
 47 		if (this.helpLink)
 48 			return this.helpLink.innerHTML;
 49 		return this.obj.innerHTML;
 50 		},
 51 		
 52 	//-------------------------------------------------------------------------
 53 	// a seconda dello stato definisce la classe css di un controllo
 54 	renderizza: function() 
 55 		{
 56 		this.obj.style.visibility = this.visibile ? '' : 'hidden';
 57 		this.obj.disabled = this.solaLettura;
 58 		this.obj.className = (this.solaLettura ? "wamodulo_disabilitato" : '') +
 59 							(this.obbligatorio ? (this.solaLettura ? " " : '') + "wamodulo_obbligatorio" : '');
 60 		}
 61 		
 62 	
 63 	}
 64 );
