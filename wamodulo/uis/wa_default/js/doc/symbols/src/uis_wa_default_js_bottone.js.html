  4 /**
  5 * classe wabottone: contiene attributi e metodi relativi a un bottone HTML
  6 * 
  7 * 
  8 * @class wabottone
  9 * @extends wacontrollo
 10 */
 11 var wabottone = new Class
 12 (
 13 	{
 14 	//-------------------------------------------------------------------------
 15 	// extends
 16 	Extends: wacontrollo,
 17 
 18 	//-------------------------------------------------------------------------
 19 	// proprieta'
 20 	tipo		: 'bottone',
 21 	
 22 	/**
 23 	 * flag che indica che la pressione di questo bottone richiede abort
 24 	 * della sessione di editing sul modulo
 25 	 * @type boolean
 26 	 */
 27 	annulla		: false,	// e'un bottone di abort
 28 	
 29 	/**
 30 	 * flag che indica che la pressione di questo bottone richiede al server
 31 	 * l'eliminazione del record attualmente in editing
 32 	 * @type boolean
 33 	 */
 34 	elimina		: false,	// e' un bottone che richiede l'eliminazione di un record
 35 	myonclick	: function onclick(event) {this.form.wamodulo.controlli[this.name].alClick(event);},
 36 	onclick		: false,
 37 	
 38 	//-------------------------------------------------------------------------
 39 	//initialization
 40 	initialize: function(modulo, nome, valore, visibile, solaLettura, obbligatorio) 
 41 		{
 42 		// definizione iniziale delle proprieta'
 43 		this.parent(modulo, nome, valore, visibile, solaLettura, obbligatorio);
 44 		if (!this.obj)
 45 			// pugnuetta di safari coi button....
 46 			this.obj = this.modulo.obj.children(this.nome);
 47 			
 48 		this.annulla = this.obj.annulla;
 49 		this.elimina = this.obj.elimina;
 50 		this.obj.onclick = this.myonclick;
 51 			
 52 		},
 53 		
 54 	//-------------------------------------------------------------------------
 55 	alClick: function(event) 
 56 		{
 57 		this.modulo.tipoInvioAnnulla = this.annulla == 1;
 58 		this.modulo.tipoInvioElimina = this.elimina == 1;
 59 		this.eventoApplicazione(event, "onclick");
 60 		}
 61 		
 62 	
 63 	}
 64 );
