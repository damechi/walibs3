563 var placeX, placeY;

566 }else{
567 winoffset=(ie4)? oframe.document.body.scrollLeft : oframe.pageXOffset;

569 if(ns4)iwidth=oframe.innerWidth;// was screwed in mozilla, fixed now?

573 ohpos=7;

575 ohpos=8;
576 }
577 }
578 if(ohpos==9){// Center
579 placeX=ox+ooffsetx-(owidth/2);

581 if(ohpos==8){// Right
582 placeX=ox+ooffsetx;
583 if((eval(placeX)+ eval(owidth))>(winoffset + iwidth)){
584 placeX=iwidth + winoffset - owidth;
585 if(placeX < 0)placeX=0;

587 }

593 var snapping=placeX % osnapx;
594 if(ohpos==7){
595 placeX=placeX -(osnapx + snapping);
596 }else{
597 placeX=placeX +(osnapx - snapping);
598 }
599 if(placeX < winoffset)placeX=winoffset;
600 }
601 }
