  4 /**
  5 * classe wacaptcha: controllo di input di un codice di controllo captcha
  6 * 
  7 * @class wacaptcha
  8 * @extends wacontrollo
  9 */
 10 var wacaptcha = new Class
 11 (
 12 	{
 13 	//-------------------------------------------------------------------------
 14 	// extends
 15 	Extends: wacontrollo,
 16 
 17 	//-------------------------------------------------------------------------
 18 	// proprieta'
 19 	tipo: 'captcha',
 20 	imgObj: null,		// e' l'immagine contenent la chiave
 21 		
 22 	//-------------------------------------------------------------------------
 23 	//initialization
 24 	initialize: function(modulo, nome, valore, visibile, solaLettura, obbligatorio) 
 25 		{
 26 		// definizione iniziale delle proprieta'
 27 		this.parent(modulo, nome, valore, visibile, solaLettura, obbligatorio);
 28 		this.imgObj = document.getElementById(this.modulo.nome + "_captcha_img_" + this.nome);
 29 		
 30 		},
 31 		
 32 	//-------------------------------------------------------------------------
 33 	// a seconda dello stato definisce la classe css di un controllo
 34 	renderizza: function() 
 35 		{
 36 		// oggetto principale
 37 		this.parent();
 38 
 39 		// immagine
 40 		this.imgObj.style.visibility = this.visibile ? '' : 'hidden'; 
 41 		}
 42 		
 43 
 44 	}
 45 );
