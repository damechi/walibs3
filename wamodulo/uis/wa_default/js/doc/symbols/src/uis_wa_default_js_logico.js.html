  4 /**
  5 * classe walogico: input HTML di tipo si/no, renderizzato con un checkbox.
  6 * 
  7 * Il controllo applicativo assume sempre i valori 0/1
  8 * 
  9 * @class walogico
 10 * @extends wacontrollo
 11 */
 12 var walogico = new Class
 13 (
 14 	{
 15 	//-------------------------------------------------------------------------
 16 	// extends
 17 	Extends: wacontrollo,
 18 
 19 	//-------------------------------------------------------------------------
 20 	// proprieta'
 21 	tipo: 'logico',
 22 	
 23 	//-------------------------------------------------------------------------
 24 	dammiValore: function() 
 25 		{
 26 		return this.obj.checked ? 1 : 0;
 27 		},
 28 		
 29 	//-------------------------------------------------------------------------
 30 	// inserisce un valore logico nel controllo
 31 	mettiValore: function(valore) 
 32 		{
 33 		this.obj.checked = valore ? 1 : 0;
 34 		},
 35 		
 36 	//-------------------------------------------------------------------------
 37 	verificaObbligo: function() 
 38 		{
 39 		if (!this.obbligatorio)
 40 			return true;
 41 		return this.obj.checked;
 42 		}
 43 		
 44 	}
 45 );
