  4 /**
  5 * classe wanoncontrollo: elemento arbitrario all'interno di un modulo
  6 * 
  7 * @class wanoncontrollo
  8 * @extends wacontrollo
  9 */
 10 var wanoncontrollo = new Class
 11 (
 12 	{
 13 	//-------------------------------------------------------------------------
 14 	// extends
 15 	Extends: wacontrollo,
 16 
 17 	//-------------------------------------------------------------------------
 18 	// proprieta'
 19 	tipo: 'noncontrollo',
 20 	
 21 	//-------------------------------------------------------------------------
 22 	//initialization
 23 	initialize: function(modulo, nome, valore, visibile, solaLettura, obbligatorio) 
 24 		{
 25 		// definizione iniziale delle proprieta'
 26 		this.parent(modulo, nome, valore, visibile, solaLettura, obbligatorio);
 27 		this.obj = document.getElementById(nome);
 28 		
 29 		},
 30 		
 31 	//-------------------------------------------------------------------------
 32 	/**
 33 	* inserisce un valore applicativo nel controllo
 34 	* 
 35 	* @param {mixed} valore valore da inserire nel controllo
 36 	*/
 37 	mettiValore: function(valore) 
 38 		{
 39 		this.obj.innerHTML = valore;
 40 		},
 41 		
 42 	//-------------------------------------------------------------------------
 43 	/**
 44 	* restituisce il  valore applicativo contenuto nel controllo
 45 	*/
 46 	dammiValore: function() 
 47 		{
 48 		return this.obj.innerHTML;
 49 		},
 50 		
 51 	//-------------------------------------------------------------------------
 52 	/**
 53 	* a seconda dello stato /definisce la classe css di un controllo(visibile, 
 54 	* solaLettura, obbligatorio) renderizza il controllo
 55 	*/
 56 	renderizza: function() 
 57 		{
 58 		this.parent();
 59 		this.obj.className = "wamodulo_noncontrollo" + (this.obbligatorio ? " wamodulo_obbligatorio" : '');
 60 		}
 61 		
 62 	
 63 	
 64 	}
 65 );
