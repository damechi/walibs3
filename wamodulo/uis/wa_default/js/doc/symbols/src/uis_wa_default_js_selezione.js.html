  4 /**
  5 * classe waselezione: select con possibilità di selezione singola
  6 * 
  7 * @class waselezione
  8 * @extends wacontrollo
  9 */
 10 var waselezione = new Class
 11 (
 12 	{
 13 	//-------------------------------------------------------------------------
 14 	// extends
 15 	Extends: wacontrollo,
 16 
 17 	//-------------------------------------------------------------------------
 18 	// proprieta'
 19 	tipo: 'selezione',
 20 	
 21 	//-------------------------------------------------------------------------
 22 	/**
 23 	 * svuota il controllo di tutte le sue opzioni
 24 	 */
 25 	svuota: function() 
 26 		{
 27 		var ultimaOptDaEliminare = 0;
 28 		if (this.obj.options[0])
 29 			{
 30 			if (this.obj.options[0].value == '')
 31 				ultimaOptDaEliminare = 1;
 32 			}
 33 		for (var i = this.obj.options.length - 1; i >= ultimaOptDaEliminare; i--)
 34 			this.obj.options[i] = null;
 35 		this.obj.selectedIndex = 0;
 36 		
 37 		return this.obj;
 38 		},
 39 		
 40 	//-------------------------------------------------------------------------
 41 	/**
 42 	 * riempie il controllo con le opzioni date
 43 	 * 
 44 	 * @param {object} values dizionario delle opzioni; la chiave di ogni 
 45 	 * elemento del dizionario sarà il valore dell'opzione; il valore di ogni
 46 	 * elemento sarà il testo dell'opzione
 47 	 * @param {string} toSelect eventuale valore dell'opzione da rendere attiva 
 48 	 * (selezionata)
 49 	 */
 50 	riempi: function(values, toSelect) 
 51 		{
 52 		var toret = false;
 53 		for (var key in values)
 54 			{
 55 			if (values[key] != undefined && values[key] != 'undefined')
 56 				{
 57 				this.obj.options[this.obj.options.length] = new Option(values[key], key);
 58 				toret = values[key] == toSelect;
 59 				}
 60 			}
 61 		this.obj.value = toSelect;
 62 		return toret;
 63 			
 64 		}
 65 		
 66 	}
 67 );
