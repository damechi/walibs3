1 //*****************************************************************************
  2 //*****************************************************************************
  3 //*****************************************************************************
  4 /**
  5 * classe wadataora: controllo composito per l'input di una data/ora
  6 * 
  7 * @class wadata
  8 * @extends wadata
  9 */
 10 var wadataora = new Class
 11 (
 12 	{
 13 	//-------------------------------------------------------------------------
 14 	// extends
 15 	Extends: wadata,
 16 
 17 	//-------------------------------------------------------------------------
 18 	// proprieta'
 19 	tipo: 'dataora',
 20 	
 21 	/**
 22 	 * oggetto HTML fisico utilizzato per selezionare l'ora
 23 	 * @type HTML_object
 24 	 */
 25 	selezioneOra: null,
 26 	
 27 	/**
 28 	 * oggetto HTML fisico utilizzato per selezionare i minuti
 29 	 * @type HTML_object
 30 	 */
 31 	selezioneMin: null,
 32 	
 33 	/**
 34 	 * oggetto HTML fisico utilizzato per selezionare i secondi
 35 	 * @type HTML_object
 36 	 */
 37 	selezioneSec: null,
 38 	
 39 	//-------------------------------------------------------------------------
 40 	//initialization
 41 	initialize: function(modulo, nome, valore, visibile, solaLettura, obbligatorio) 
 42 		{
 43 		// definizione iniziale delle proprieta'
 44 		this.parent(modulo, nome, valore, visibile, solaLettura, obbligatorio);
 45 		this.selezioneOra = this.modulo.obj.elements["wamodulo_ora_" + this.nome];
 46 		this.selezioneMin = this.modulo.obj.elements["wamodulo_min_" + this.nome];
 47 		this.selezioneSec = this.modulo.obj.elements["wamodulo_sec_" + this.nome];
 48 		},
 49 		
 50 	//-------------------------------------------------------------------------
 51 	// restituisce il valore logico del controllo
 52 	dammiValore: function() 
 53 		{
 54 		if (this.selezioneGiorno.value == '' || this.selezioneMese.value == '' || this.selezioneAnno == '' ||
 55 			this.selezioneOra.value == '' || this.selezioneMin.value == '' ||
 56 			(this.selezioneSec && this.selezioneSec.value == '')
 57 			)
 58 			return false;
 59 		return new Date(this.selezioneAnno.value, this.selezioneMese.value - 1, this.selezioneGiorno.value,
 60 						this.selezioneOra.value, this.selezioneMin.value, this.selezioneSec ? this.selezioneSec.value : 0);
 61 		},
 62 		
 63 	//-------------------------------------------------------------------------
 64 	// inserisce un valore logico nel controllo
 65 	mettiValore: function(valore) 
 66 		{
 67 		this.parent(valore);
 68 		if (!valore || valore == 0 || valore == '')
 69 			{
 70 			this.selezioneOra.value = this.selezioneMin.value = '';
 71 			if (this.selezioneSec)
 72 				this.selezioneSec.value = '';
 73 			}
 74 		else
 75 			{
 76 			this.selezioneOra.value = valore.getHours(); 
 77 			this.selezioneMin.value = valore.getMinutes(); 
 78 			if (this.selezioneSec)
 79 				this.selezioneSec.value = valore.getSeconds();
 80 			}
 81 		},
 82 		
 83 	//-------------------------------------------------------------------------
 84 	verificaForma: function() 
 85 		{
 86 		if (!this.dammiValore())
 87 			return true;
 88 			
 89 		var testDate = new Date(this.selezioneAnno.value, this.selezioneMese.value - 1, this.selezioneGiorno.value,
 90 						this.selezioneOra.value, this.selezioneMin.value, this.selezioneSec ? this.selezioneSec.value : 0);
 91 		return this.selezioneGiorno.value == testDate.getDate() &&
 92 					this.selezioneMese.value == (testDate.getMonth() + 1) &&
 93 					this.selezioneAnno.value == testDate.getFullYear() &&
 94 					this.selezioneOra.value == testDate.getHours() &&
 95 					this.selezioneMin.value == testDate.getMinutes() &&
 96 					(this.selezioneSec ? this.selezioneSec.value == testDate.getSeconds() : true);
 97 		},
 98 		
 99 	//-------------------------------------------------------------------------
100 	// a seconda dello stato definisce la classe css di un controllo
101 	renderizza: function() 
102 		{
103 		this.parent();
104 		
105 		this.selezioneOra.style.visibility = this.selezioneMin.style.visibility = this.visibile ? '' : 'hidden';
106 		if (this.selezioneSec)
107 			this.selezioneSec.style.visibility = this.selezioneOra.style.visibility;
108 		
109 		this.selezioneOra.disabled = this.selezioneMin.disabled = this.solaLettura;
110 		if (this.selezioneSec)
111 			this.selezioneSec.disabled = this.selezioneOra.disabled;
112 		
113 		this.selezioneOra.className = this.selezioneMin.className = 
114 								(this.obbligatorio ? "wamodulo_obbligatorio" : '');
115 		if (this.selezioneSec)
116 			this.selezioneSec.className = this.selezioneOra.className;
117 		},
118 		
119 	//-------------------------------------------------------------------------
120 	// associa un evento al controllo
121 	aggiungiEvento: function (nomeEvento, evento)
122 		{
123 		this.parent(nomeEvento, evento);
124 		this.selezioneOra[nomeEvento] = evento;
125 		this.selezioneMin[nomeEvento] = evento;
126 		if (this.selezioneSec)
127 			this.selezioneSec[nomeEvento] = evento;
128 		},
129 	
130 	//-------------------------------------------------------------------------
131 	// verifica se un controllo fisico appartiene al controllo logico
132 	miAppartiene: function (obj)
133 		{
134 		if  (this.parent(obj) ||
135 				obj.name == this.selezioneOra.name ||
136 				obj.name == this.selezioneMin.name)
137 			return true;
138 		if (this.selezioneSec)
139 			return obj.name == this.selezioneSec.name;
140 		return false;
141 		}
142 	
143 	}
144 );
