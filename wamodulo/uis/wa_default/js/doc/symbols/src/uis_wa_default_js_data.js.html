1 //*****************************************************************************
  2 //*****************************************************************************
  3 //*****************************************************************************
  4 /**
  5 * classe wadata: controllo composito per l'input di una data
  6 * 
  7 * @class wadata
  8 * @extends wacontrollo
  9 */
 10 var wadata = new Class
 11 (
 12 	{
 13 	//-------------------------------------------------------------------------
 14 	// extends
 15 	Extends: wacontrollo,
 16 
 17 	//-------------------------------------------------------------------------
 18 	// proprieta'
 19 	tipo			: 'data',
 20 	
 21 	/**
 22 	 * oggetto HTML fisico utilizzato per contenere il calendario
 23 	 * @type HTML_object
 24 	 */
 25 	divCalendario	: null,
 26 	
 27 	/**
 28 	 * oggetto HTML fisico utilizzato per selezionare il giorno
 29 	 * @type HTML_object
 30 	 */
 31 	selezioneGiorno	: null,
 32 	
 33 	/**
 34 	 * oggetto HTML fisico utilizzato per selezionare il mese
 35 	 * @type HTML_object
 36 	 */
 37 	selezioneMese	: null,
 38 	
 39 	/**
 40 	 * oggetto HTML fisico utilizzato per selezionare l'anno
 41 	 * @type HTML_object
 42 	 */
 43 	selezioneAnno	: null,
 44 	
 45 	/**
 46 	 * oggetto HTML fisico utilizzato per mostrare il calendario mensile
 47 	 * @type HTML_object
 48 	 */
 49 	bottoneCalMese	: null,
 50 	
 51 	/**
 52 	 * oggetto HTML fisico utilizzato per mostrare il calendario annuale
 53 	 * @type HTML_object
 54 	 */
 55 	bottoneCalAnno	: null,
 56 	
 57 	//-------------------------------------------------------------------------
 58 	//initialization
 59 	initialize: function(modulo, nome, valore, visibile, solaLettura, obbligatorio) 
 60 		{
 61 		// definizione iniziale delle proprieta'
 62 		this.parent(modulo, nome, valore, visibile, solaLettura, obbligatorio);
 63 		this.selezioneGiorno = this.modulo.obj.elements["wamodulo_giorno_" + this.nome];
 64 		this.selezioneMese = this.modulo.obj.elements["wamodulo_mese_" + this.nome];
 65 		this.selezioneAnno = this.modulo.obj.elements["wamodulo_anno_" + this.nome];
 66 		this.bottoneCalMese = this.modulo.obj.elements["wamodulo_mesecal_" + this.nome];
 67 		this.bottoneCalAnno = this.modulo.obj.elements["wamodulo_annocal_" + this.nome];
 68 		this.divCalendario = document.getElementById("overDiv");
 69 		if (!this.divCalendario)
 70 			{
 71 			this.divCalendario = document.createElement('div');
 72    			this.divCalendario.setAttribute('id', "overDiv");
 73    			this.divCalendario.setAttribute('class', "calpop");
 74    			document.body.appendChild(this.divCalendario);
 75 			}
 76 		},
 77 		
 78 	//-------------------------------------------------------------------------
 79 	// restituisce il valore logico del controllo
 80 	dammiValore: function() 
 81 		{
 82 		if (this.selezioneGiorno.value == '' || this.selezioneMese.value == '' || this.selezioneAnno.value == '')
 83 			return false;
 84 //		return new Date(this.selezioneAnno.value, this.selezioneMese.value - 1, this.selezioneGiorno.value);
 85 		var retval = new Date();
 86 		retval.setUTCFullYear(this.selezioneAnno.value, this.selezioneMese.value - 1, this.selezioneGiorno.value);
 87 		return retval;
 88 		},
 89 		
 90 	//-------------------------------------------------------------------------
 91 	// inserisce un valore logico nel controllo
 92 	mettiValore: function(valore) 
 93 		{
 94 		if (!valore || valore == 0 || valore == '')
 95 			this.selezioneGiorno.value = this.selezioneMese.value = this.selezioneAnno.value = '';
 96 		else
 97 			{
 98 			this.selezioneGiorno.value = valore.getUTCDate(); 
 99 			this.selezioneMese.value = valore.getUTCMonth() + 1; 
100 			this.selezioneAnno.value = valore.getUTCFullYear();
101 			}
102 		},
103 		
104 	//-------------------------------------------------------------------------
105 	verificaForma: function() 
106 		{
107 		if (!this.dammiValore())
108 			return true;
109 			
110 //		var testDate = new Date(this.selezioneAnno.value, this.selezioneMese.value - 1, this.selezioneGiorno.value);
111 		var testDate = new Date();
112 		testDate.setUTCFullYear(this.selezioneAnno.value, this.selezioneMese.value - 1, this.selezioneGiorno.value);
113 		return this.selezioneGiorno.value == testDate.getUTCDate() &&
114 					this.selezioneMese.value == (testDate.getUTCMonth() + 1) &&
115 					this.selezioneAnno.value == testDate.getUTCFullYear();
116 		},
117 		
118 	//-------------------------------------------------------------------------
119 	// a seconda dello stato definisce la classe css di un controllo
120 	renderizza: function() 
121 		{
122 		this.selezioneGiorno.style.visibility = this.selezioneMese.style.visibility = this.selezioneAnno.style.visibility = this.visibile ? '' : 'hidden';
123 		if (this.bottoneCalMese)
124 			this.bottoneCalMese.style.visibility = this.selezioneGiorno.style.visibility;
125 		if (this.bottoneCalAnno)
126 			this.bottoneCalAnno.style.visibility = this.selezioneGiorno.style.visibility;
127 
128 		this.selezioneGiorno.disabled = this.selezioneMese.disabled = this.selezioneAnno.disabled = this.solaLettura;
129 		if (this.bottoneCalMese)
130 			this.bottoneCalMese.disabled = this.selezioneGiorno.disabled;
131 		if (this.bottoneCalAnno)
132 			this.bottoneCalAnno.disabled = this.selezioneGiorno.disabled;
133 			
134 		this.selezioneGiorno.className = 
135 			this.selezioneMese.className = 
136 			this.selezioneAnno.className = (this.obbligatorio ? "wamodulo_obbligatorio" : '');
137 								
138 		},
139 		
140 	//-------------------------------------------------------------------------
141 	// associa un evento al controllo
142 	aggiungiEvento: function (nomeEvento, evento)
143 		{
144 		this.selezioneGiorno[nomeEvento] = evento;
145 		this.selezioneMese[nomeEvento] = evento;
146 		this.selezioneAnno[nomeEvento] = evento;
147 		},
148 	
149 	//-------------------------------------------------------------------------
150 	// verifica se un controllo fisico appartiene al controllo logico
151 	miAppartiene: function (obj)
152 		{
153 		return obj.name == this.selezioneGiorno.name ||
154 				obj.name == this.selezioneMese.name ||
155 				obj.name == this.selezioneAnno.name;
156 		},
157 	
158 	//-------------------------------------------------------------------------
159 	// simula un evento sul controllo
160 	simulaEvento: function(tipoEvento) 
161 		{
162 		var event;
163 		if (document.createEvent) 
164 			{
165 			event = document.createEvent("HTMLEvents");
166 			event.initEvent(tipoEvento, true, true);
167 			this.selezioneGiorno.dispatchEvent(event);
168 			} 
169 		else 
170 			{
171 			event = document.createEventObject();
172 			event.eventType = tipoEvento;
173 			this.selezioneGiorno.fireEvent("on" + event.eventType, event);
174 			}
175 		
176 		}
177 		
178 	
179 	
180 	}
181 );
